Paddlex使用笔记

环境准备

1）安装paddlepaddle

python -m pip install paddlepaddle-gpu==2.2.2.post101 -f https://www.paddlepaddle.org.cn/whl/linux/mkl/avx/stable.html

2）安装paddlex

pip install paddlex -i https://mirror.baidu.com/pypi/simple

训练历程

1）定义图像的预处理和数据增强操作

from paddlex.det import transforms
train_transforms=transforms.Compose([
    transforms.MixupImage(mixup_epoch=250),#图像混合
    transforms.RandomDistort(),#随即像素变换
    transforms.RandomExpand(),#随机膨胀
    transforms.RandomCrop(),#随即裁剪
    transforms.Resize(target_size=512,interp='RANDOM'),
    transforms.RandomHorizontalFlip(),#随机水平翻转
    transforms.Normalize(),
])
eval_transforms=transforms.Compose([
    transforms.Resize(target_size=512,interp='CUBIC'),
    transforms.Normalize(),
])

2）定义数据集(VOC格式)

base='./work/pascalvoc/VOCdevkit/VOC2012'
imgs=os.listdir('./work/pascalvoc/VOCdevkit/VOC2012/JPEGImages')

with open(os.path.join(base,'train_list.txt'),'w') as f:
    for im in imgs[:-200]:
        info='JPEGImages/'+im+' '
        info+='Annotations/'+im[:-4]+'.xml\n'

with open(os.path.join(base,'val_list.txt'),'w') as f:
    for im in imgs[-200:]:
        info='JPEGImages/'+im+' '
        info+='Annotations/'+im[:-4]+'.xml\n'
        f.write(info)

CLASSES=['aeroplane','bicycle','bird','boat','bottle','bus','car','cat','chair','cow',
         'diningtable','dog','horse','motorbike','person','pottedplant','sheep','sofa',
         'train','tvmonitor']

with open('labels.txt','w') as f:
    for v in CLASSES:
        f.write(v+'\n')

3）加载数据集

train_dataset=pdx.datasets.VOCDetection(
    data_dir=base,
    file_list=os.path.join(base,'train_list.txt'),
    label_list='label.txt',
    transforms=train_transforms,
    shuffle=True
)

eval_dataset=pdx.datasets.VOCDetection(
    data_dir=base,
    file_list=os.path.join(base,'val_list.txt'),
    label_list='labels.txt',
    transforms=eval_transforms,
)

4）定义模型开始训练

model=pdx.det.YOLOv3(
    num_classes=num_classes,
    backbone='DarkNet53'
)
model.train(
    num_epochs=20,
    train_dataset=train_dataset,
    train_batch_size=4,
    eval_dataset=eval_dataset,
    learning_rate=0.00025,
    lr_decay_epochs=[10,15],
    save_internal_epochs=4,
    log_internal_steps=100,
    save_dir='./YOLOv3',
    pretrain_weights='IMAGENET',
    use_vdl=True
)

5）加载模型

model=pdx.load_model('./YOLOv3/best_model')

6）评估模型

model.evaluate(eval_dataset,batch_size=1,epoch_id=None,metric=None,return_details=False)

7）模型预测

result = model.predict(test_jpg)
